# Compiler Build Commands
# --------------------------
# Run these commands in order to build the compiler

# IMPORTANT: Use WinFlexBison (in win_flex_bison folder)
# Do NOT use the system bison/flex - they have path issues!

# Step 1: Generate parser with BISON
.\win_flex_bison\win_bison.exe -d 1.y

# Step 2: Generate lexer with FLEX
.\win_flex_bison\win_flex.exe 1.l

# Step 3: Compile everything together
gcc -o compiler.exe 1.tab.c lex.yy.c SyntaxTree.c

# Step 4: Run the compiler
.\compiler.exe

# --------------------------
# Quick one-liner for building:
# .\win_flex_bison\win_bison.exe -d 1.y; .\win_flex_bison\win_flex.exe 1.l; gcc -o compiler.exe 1.tab.c lex.yy.c SyntaxTree.c

# --------------------------
# Output files after running:
# - out.txt: Contains target code
# - tree.txt: Contains syntax tree visualization
# - outError.txt: Contains any compilation errors
# - temp: Contains token log from scanner
